官网配置参考：https://www.elastic.co/guide/en/logstash/7.10/plugins-inputs-elasticsearch.html

```bash
input {
#  beats {
#    port => 5044
#  }
  elasticsearch {
    hosts => ["http://192.168.0.32:9200","http://192.168.0.33:9200","http://192.168.0.34:9200"]
    index => "exhibition.*"
#    query => '{ "query": { "query_string": { "query": "*" } } }'
    user => "elastic"
    password => "123456"
    size => 10000
    scroll => "5m"
    docinfo => true		# 设置为true，它可以通过添加到元数据的字段进行动态分配
    add_field => {		# 增加字段
    	identifier => %{[@metadata][_index]}:%{[@metadata][_type]}:%{[@metadata][_id]}"
    }
  }
}

output {
  elasticsearch {
    hosts => ["http://192.168.0.21:9200"]
#    index => "company_v4"
    index => "%{[@metadata][_index]}"
    user => "elastic"
    password => "123456"
    document_id => "%{[@metadata][_id]}"  #引用元数据的_id为主键，否则自动生成
  }
}
```
